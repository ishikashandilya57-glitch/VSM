function getProcessStatus(ocNo) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName(SHEET_VSM_EXECUTION);
    
    if (!sheet) {
      return {
        success: false,
        error: 'VSM_Execution sheet not found'
      };
    }

    const lastRow = sheet.getLastRow();
    if (lastRow < 2) {
      return {
        success: true,
        data: []
      };
    }

    const dataRange = sheet.getRange(2, 1, lastRow - 1, 14);
    const allData = dataRange.getValues();

    const processData = [];
    
    for (let i = 0; i < allData.length; i++) {
      const row = allData[i];
      const rowOcNo = row[COL.OC_NO - 1];
      
      if (rowOcNo === ocNo) {
        const processStage = row[COL.PROCESS_STAGE - 1];
        const processSeq = row[COL.PROCESS_SEQ - 1];
        const actualStartDate = row[COL.ACTUAL_START - 1];
        const actualEndDate = row[COL.ACTUAL_END - 1];
        const processStatus = row[COL.PROCESS_STATUS - 1];
        
        processData.push({
          processStage: processStage || '',
          processSeq: processSeq || 0,
          actualStartDate: actualStartDate ? formatDate(new Date(actualStartDate)) : '',
          actualEndDate: actualEndDate ? formatDate(new Date(actualEndDate)) : '',
          processStatus: processStatus || ''
        });
      }
    }

    processData.sort((a, b) => a.processSeq - b.processSeq);

    return {
      success: true,
      data: processData
    };

  } catch (error) {
    Logger.log('Error in getProcessStatus: ' + error.toString());
    return {
      success: false,
      error: error.toString()
    };
  }
}
